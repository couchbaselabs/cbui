Declaring a flex property for a flex item

          
  /* 0 0 auto */
  flex: none;
  
  /* One value, unitless number: flex-grow */
  flex: 2;
  
  /* One value, width/height: flex-basis */
  flex: 10em;
  flex: 30px;
  flex: auto;
  flex: content;
  
  /* Two values: flex-grow | flex-basis */
  flex: 1 30px;
  
  /* Two values: flex-grow | flex-shrink */
  flex: 2 2;
  
  /* Three values: flex-grow | flex-shrink | flex-basis */
  flex: 2 2 10%;
  
  /* Global values */
  flex: inherit;
  flex: initial;
  flex: unset;
